function addButtons(list,gridX,gridY,gridSpacing,xstart,ystart,buttonColor,textColor,buttonWidth,buttonHeight)
	for i=1,#list do
		BA.makeButton(xstart+((buttonWidth+gridSpacing)*((i-i%gridY-1)/gridY)),ystart+((buttonHeight+gridSpacing)*(i%gridY-1)),buttonColor,textColor,-1,-1,tostring(list[i]))
	end
end

function addRect(x1,y1,x2,y2,fillColor,borderColor,borderSize)
	gpu.setBackground(borderColor)
	gpu.fill(x1,y1,x2-x1,y2-y1," ")
	gpu.setBackground(fillColor)
	gpu.fill(x1+borderSize,y1+borderSize,x2-x1-borderSize,y2-y1-borderSize," ")
end

function addText(x,y,str,textColor,backgroundColor)
	gpu.setForeground(textColor)
	gpu.setBackground(backgroundColor)
	gpu.set(x,y,str)
end

function addTextCentered(y,x1,x2,str,textColor,backgroundColor)
	--centeres text between x1 and x2
	xcenter = (x1+x2)/2
	xtext = xcenter-(len(tostring(str))/2)
	addText(xtext,y,str,textColor,backgroundColor)
end
